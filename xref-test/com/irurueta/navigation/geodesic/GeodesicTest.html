987     private static boolean isNaN(double x) {
988         //noinspection ExpressionComparedToItself
989         return x != x;

992     private static PolygonResult planimeter(final double[][] points) {
993         POLYGON.clear();
994         for (final var point : points) {
995             POLYGON.addPoint(point[0], point[1]);
996         }
997         return POLYGON.compute(false, true);
998     }
999 
1000     private static PolygonResult polyLength(final double[][] points) {
1001         POLYLINE.clear();
1002         for (final var point : points) {
1003             POLYLINE.addPoint(point[0], point[1]);
1004         }
1005         return POLYLINE.compute(false, true);
1006     }
