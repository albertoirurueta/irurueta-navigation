69      @Test
70      void testEstimate() throws AlgebraException {
71          for (var t = 0; t < TIMES; t++) {
72              final var randomizer = new UniformRandomizer();
73  
74              final var userLatitude = Math.toRadians(randomizer.nextDouble(MIN_LATITUDE_DEGREES, MAX_LATITUDE_DEGREES));
75              final var userLongitude = Math.toRadians(randomizer.nextDouble(MIN_LONGITUDE_DEGREES,
76                      MAX_LONGITUDE_DEGREES));
77              final var userHeight = randomizer.nextDouble(MIN_HEIGHT_METERS, MAX_HEIGHT_METERS);
78  
79              final var userVn = randomizer.nextDouble(MIN_SPEED_VALUE, MAX_SPEED_VALUE);
80              final var userVe = randomizer.nextDouble(MIN_SPEED_VALUE, MAX_SPEED_VALUE);
81              final var userVd = randomizer.nextDouble(MIN_SPEED_VALUE, MAX_SPEED_VALUE);
82  
83              final var userNedFrame = new NEDFrame(userLatitude, userLongitude, userHeight, userVn, userVe, userVd);
84              final var userEcefFrame = NEDtoECEFFrameConverter.convertNEDtoECEFAndReturnNew(userNedFrame);
85  
86              final var userPosition = userEcefFrame.getPosition();
87  
88              final var numMeasurements = randomizer.nextInt(MIN_MEASUREMENTS, MAX_MEASUREMENTS);
89  
90              final var measurements = new ArrayList<GNSSMeasurement>();
91              for (var i = 0; i < numMeasurements; i++) {
92                  final var satLatitude = Math.toRadians(randomizer.nextDouble(MIN_LATITUDE_DEGREES,
93                          MAX_LATITUDE_DEGREES));
94                  final var satLongitude = Math.toRadians(randomizer.nextDouble(MIN_LONGITUDE_DEGREES,
95                          MAX_LONGITUDE_DEGREES));
96                  final var satHeight = randomizer.nextDouble(MIN_SAT_HEIGHT_METERS,
97                          MAX_SAT_HEIGHT_METERS);
98  
99                  final var satVn = randomizer.nextDouble(MIN_SAT_SPEED_VALUE, MAX_SAT_SPEED_VALUE);
100                 final var satVe = randomizer.nextDouble(MIN_SAT_SPEED_VALUE, MAX_SAT_SPEED_VALUE);
101                 final var satVd = randomizer.nextDouble(MIN_SAT_SPEED_VALUE, MAX_SAT_SPEED_VALUE);
102 
103                 final var satNedFrame = new NEDFrame(satLatitude, satLongitude, satHeight, satVn, satVe, satVd);
104                 final var satEcefFrame = NEDtoECEFFrameConverter.convertNEDtoECEFAndReturnNew(satNedFrame);
105 
106                 final var satPosition = satEcefFrame.getPosition();
107 
108                 final var pseudoRange = userPosition.distanceTo(satPosition);
109 
110                 final var posDiff = new double[]{
111                         satEcefFrame.getX() - userEcefFrame.getX(),
112                         satEcefFrame.getY() - userEcefFrame.getY(),
113                         satEcefFrame.getZ() - userEcefFrame.getZ()};

117                         satEcefFrame.getVx() - userEcefFrame.getVx(),
118                         satEcefFrame.getVy() - userEcefFrame.getVy(),
119                         satEcefFrame.getVz() - userEcefFrame.getVz()};

126 
127                 final var x = satEcefFrame.getX();
128                 final var y = satEcefFrame.getY();
129                 final var z = satEcefFrame.getZ();
130 
131                 final var vx = satEcefFrame.getVx();
132                 final var vy = satEcefFrame.getVy();
133                 final var vz = satEcefFrame.getVz();

135                 measurements.add(new GNSSMeasurement(pseudoRange, pseudoRate, x, y, z, vx, vy, vz));
136             }

139                     userEcefFrame.getX(), userEcefFrame.getY(), userEcefFrame.getZ(),
140                     userEcefFrame.getVx(), userEcefFrame.getVy(), userEcefFrame.getVz(), CLOCK_OFFSET, CLOCK_DRIFT);
141

147             final var initialClockOffsetUncertainty = randomizer.nextDouble(MIN_CONFIG_VALUE, MAX_CONFIG_VALUE);

151             final var clockPhasePSD = randomizer.nextDouble(MIN_CONFIG_VALUE, MAX_CONFIG_VALUE);
152             final var pseudoRangeSD = randomizer.nextDouble(MIN_CONFIG_VALUE, MAX_CONFIG_VALUE);
153             final var rangeRateSD = randomizer.nextDouble(MIN_CONFIG_VALUE, MAX_CONFIG_VALUE);
154 
155             final var config = new GNSSKalmanConfig(initialPositionUncertainty, initialVelocityUncertainty,
156                     initialClockOffsetUncertainty, initialClockDriftUncertainty, accelerationPSD, clockFrequencyPSD,
157                     clockPhasePSD, pseudoRangeSD, rangeRateSD);
158 
159             final var updatedEstimation1 = new GNSSEstimation();
160             final var updatedCovariance1 = new Matrix(GNSSEstimation.NUM_PARAMETERS, GNSSEstimation.NUM_PARAMETERS);

174             final var updatedCovariance3 = new Matrix(GNSSEstimation.NUM_PARAMETERS, GNSSEstimation.NUM_PARAMETERS);
175             estimate(measurements, previousEstimation, previousCovariance, config, updatedEstimation3,
176                     updatedCovariance3);
177 
178             assertNotNull(updatedEstimation1);
179

181                 continue;
182             }
183             if (Math.abs(updatedEstimation1.getY() - updatedEstimation3.getY()) > ABSOLUTE_ERROR) {
184                 continue;
185             }
186             if (Math.abs(updatedEstimation1.getZ() - updatedEstimation3.getZ()) > ABSOLUTE_ERROR) {
187                 continue;
188             }
189             assertEquals(updatedEstimation1.getX(), updatedEstimation3.getX(), ABSOLUTE_ERROR);
190             assertEquals(updatedEstimation1.getY(), updatedEstimation3.getY(), ABSOLUTE_ERROR);
191             assertEquals(updatedEstimation1.getZ(), updatedEstimation3.getZ(), ABSOLUTE_ERROR);
192 
193             assertEquals(updatedEstimation1.getVx(), updatedEstimation3.getVx(), ABSOLUTE_ERROR);
194             assertEquals(updatedEstimation1.getVy(), updatedEstimation3.getVy(), ABSOLUTE_ERROR);
195             assertEquals(updatedEstimation1.getVz(), updatedEstimation3.getVz(), ABSOLUTE_ERROR);
196 
197             assertEquals(updatedEstimation1.getClockOffset(), updatedEstimation3.getClockOffset(), ABSOLUTE_ERROR);
198             assertEquals(updatedEstimation1.getClockDrift(), updatedEstimation3.getClockDrift(), ABSOLUTE_ERROR);
199 
200             assertTrue(updatedCovariance1.equals(updatedCovariance3, ABSOLUTE_ERROR));
201 
202             assertEquals(updatedEstimation1, updatedEstimation2);
203             assertEquals(updatedCovariance1, updatedCovariance2);
204         }

343     private void estimate(final List<GNSSMeasurement> measurements,
344                           final GNSSEstimation previousEstimation,
345                           final Matrix previousCovariance,
346                           final GNSSKalmanConfig config,
347                           final GNSSEstimation updatedEstimation,
348                           final Matrix updatedCovariance) throws AlgebraException {
349 
350         // 1. Determine transition matrix using (9.147) and (9.150)
351         final var phiMatrix = Matrix.identity(8, 8);
352         phiMatrix.setElementAt(0, 3, TIME_INTERVAL_SECONDS);
353         phiMatrix.setElementAt(1, 4, TIME_INTERVAL_SECONDS);
354         phiMatrix.setElementAt(2, 5, TIME_INTERVAL_SECONDS);
355         phiMatrix.setElementAt(6, 7, TIME_INTERVAL_SECONDS);
356 
357         // 2. Determine system noise covariance matrix using (9.152)
358         final var qMatrix = new Matrix(8, 8);
359         qMatrix.setSubmatrix(0, 0, 2, 2,
360                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(config.getAccelerationPSD()
361                         * Math.pow(TIME_INTERVAL_SECONDS, 3.0) / 3.0));
362         qMatrix.setSubmatrix(0, 3, 2, 5,
363                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(config.getAccelerationPSD()
364                         * Math.pow(TIME_INTERVAL_SECONDS, 2.0) / 2.0));
365         qMatrix.setSubmatrix(3, 0, 5, 2,
366                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(config.getAccelerationPSD()
367                         * Math.pow(TIME_INTERVAL_SECONDS, 2.0) / 2.0));
368         qMatrix.setSubmatrix(3, 3, 5, 5,
369                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(config.getAccelerationPSD()
370                         * TIME_INTERVAL_SECONDS));
371         qMatrix.setElementAt(6, 6, config.getClockFrequencyPSD()
372                 * Math.pow(TIME_INTERVAL_SECONDS, 3.0) / 3.0 + config.getClockPhasePSD() * TIME_INTERVAL_SECONDS);
373         qMatrix.setElementAt(6, 7, config.getClockFrequencyPSD()
374                 * Math.pow(TIME_INTERVAL_SECONDS, 2.0) / 2.0);
375         qMatrix.setElementAt(7, 6, config.getClockFrequencyPSD()
376                 * Math.pow(TIME_INTERVAL_SECONDS, 2.0) / 2.0);
377         qMatrix.setElementAt(7, 7, config.getClockFrequencyPSD() * TIME_INTERVAL_SECONDS);
378 
379         // 3. Propagate state estimates using (3.14)
380         final var xEstOld = previousEstimation.asMatrix();
381         final var xEstPropagated = phiMatrix.multiplyAndReturnNew(xEstOld);
382 
383         // 4. Propagate state estimation error covariance matrix using (3.15)
384         final var pMatrixPropagated = phiMatrix.multiplyAndReturnNew(previousCovariance.multiplyAndReturnNew(
385                 phiMatrix.transposeAndReturnNew())).addAndReturnNew(qMatrix);
386 
387         // Measurement update phase
388 
389         // Skew symmetric matrix of Earth rate
390         final var omegaIe = Utils.skewMatrix(new double[]{0.0, 0.0, Constants.EARTH_ROTATION_RATE});
391 
392         final var noMeas = measurements.size();
393         final var uAseT = new Matrix(noMeas, 3);
394         final var predMeas = new Matrix(noMeas, 2);
395 
396         // Loop measurements
397         for (var j = 0; j < noMeas; j++) {
398             // Predict approx range
399             final var measurement = measurements.get(j);
400             var deltaR = Matrix.newFromArray(new double[]{
401                     measurement.getX() - xEstPropagated.getElementAtIndex(0),
402                     measurement.getY() - xEstPropagated.getElementAtIndex(1),
403                     measurement.getZ() - xEstPropagated.getElementAtIndex(2)
404             });
405 
406             final var approxRange = Math.sqrt(Utils.dotProduct(deltaR.transposeAndReturnNew(), deltaR));
407 
408             // Calculate frame rotation during signal transit time using (8.36)
409             final var cei = new Matrix(3, 3);
410             cei.fromArray(new double[]{
411                     1.0, Constants.EARTH_ROTATION_RATE * approxRange / GNSSKalmanEpochEstimator.SPEED_OF_LIGHT, 0.0,
412                     -Constants.EARTH_ROTATION_RATE * approxRange / GNSSKalmanEpochEstimator.SPEED_OF_LIGHT, 1.0, 0.0,
413                     0.0, 0.0, 1.0}, false);
414 
415             // Predict pseudo-range using (9.165)
416             final var measurementPosition = Matrix.newFromArray(new double[]{
417                     measurement.getX(), measurement.getY(), measurement.getZ()});
418             final var measurementVelocity = Matrix.newFromArray(new double[]{
419                     measurement.getVx(), measurement.getVy(), measurement.getVz()});
420             final var propagatedPosition = xEstPropagated.getSubmatrix(
421                     0, 0, 2, 0);
422             deltaR = cei.multiplyAndReturnNew(measurementPosition).subtractAndReturnNew(propagatedPosition);
423             final var range = Math.sqrt(Utils.dotProduct(deltaR.transposeAndReturnNew(), deltaR));
424             predMeas.setElementAt(j, 0, range + xEstPropagated.getElementAtIndex(6));
425 
426             // Predict line of sight
427             uAseT.setSubmatrix(j, 0, j, 2,
428                     deltaR.transposeAndReturnNew().multiplyByScalarAndReturnNew(1.0 / range));
429 
430             // Predict pseudo-range rate using (9.165)
431             final var rangeRate = uAseT.getSubmatrix(j, 0, j, 2).multiplyAndReturnNew(
432                     cei.multiplyAndReturnNew(measurementVelocity.addAndReturnNew(
433                             omegaIe.multiplyAndReturnNew(measurementPosition)))
434                             .subtractAndReturnNew(xEstPropagated.getSubmatrix(
435                                     3, 0, 5, 0)
436                                     .addAndReturnNew(omegaIe.multiplyAndReturnNew(xEstPropagated.getSubmatrix(
437                                                     0, 0, 2,
438                                             0)))));
439             predMeas.setElementAt(j, 1, rangeRate.getElementAtIndex(0)
440                     + xEstPropagated.getElementAtIndex(7));
441         }
442 
443         // 5. Set-up measurement matrix using (9.163)
444         final var hMatrix = new Matrix(2 * noMeas, 8);
445         hMatrix.setSubmatrix(0, 0, noMeas - 1, 2,
446                 uAseT.getSubmatrix(0, 0, noMeas - 1, 2)
447                         .multiplyByScalarAndReturnNew(-1.0));
448         hMatrix.setSubmatrix(0, 3, noMeas - 1, 5,
449                 new Matrix(noMeas, 3));
450         final var ones = new double[noMeas];
451         Arrays.fill(ones, 1.0);
452         hMatrix.setSubmatrix(0, 6, noMeas - 1, 6, ones);
453         hMatrix.setSubmatrix(0, 7, noMeas - 1, 7,
454                 new double[noMeas]);
455         hMatrix.setSubmatrix(noMeas, 0, 2 * noMeas - 1, 2,
456                 new Matrix(noMeas, 3));
457         hMatrix.setSubmatrix(noMeas, 3, 2 * noMeas - 1, 5,
458                 uAseT.getSubmatrix(0, 0, noMeas - 1, 2)
459                         .multiplyByScalarAndReturnNew(-1.0));
460         hMatrix.setSubmatrix(noMeas, 6, 2 * noMeas - 1, 6,
461                 new Matrix(noMeas, 1));
462         hMatrix.setSubmatrix(noMeas, 7, 2 * noMeas - 1, 7, ones);
463 
464         // 6. Set-up measurement noise covariance matrix assuming all measurements
465         // are independent and have equal variance for a given measurement type
466         final var rMatrix = new Matrix(2 * noMeas, 2 * noMeas);
467         rMatrix.setSubmatrix(0, 0, noMeas - 1, noMeas - 1,
468                 Matrix.identity(noMeas, noMeas).multiplyByScalarAndReturnNew(Math.pow(config.getPseudoRangeSD(), 2.0)));
469         rMatrix.setSubmatrix(0, noMeas, noMeas - 1, 2 * noMeas - 1,
470                 new Matrix(noMeas, noMeas));
471         rMatrix.setSubmatrix(noMeas, 0, 2 * noMeas - 1, noMeas - 1,
472                 new Matrix(noMeas, noMeas));
473         rMatrix.setSubmatrix(noMeas, noMeas, 2 * noMeas - 1, 2 * noMeas - 1,
474                 Matrix.identity(noMeas, noMeas).multiplyByScalarAndReturnNew(Math.pow(config.getRangeRateSD(), 2.0)));
475 
476         // 7. Calculate Kalman gain using (3.21)
477         final var kMatrix = pMatrixPropagated.multiplyAndReturnNew(hMatrix.transposeAndReturnNew().multiplyAndReturnNew(
478                 Utils.inverse(hMatrix.multiplyAndReturnNew(pMatrixPropagated).multiplyAndReturnNew(
479                         hMatrix.transposeAndReturnNew()).addAndReturnNew(rMatrix))));
480 
481         // 8. Formulate measurement innovations using (3.88)
482         final var deltaZ = new Matrix(2 * noMeas, 1);
483         for (var j = 0; j < noMeas; j++) {
484             final var measurement = measurements.get(j);
485             deltaZ.setElementAtIndex(j, measurement.getPseudoRange() - predMeas.getElementAtIndex(j));
486             deltaZ.setElementAtIndex(noMeas + j,
487                     measurement.getPseudoRate() - predMeas.getElementAtIndex(noMeas + j));
488         }
489 
490         // 9. Update state estimates using (3.24)
491         final var xEstNew = xEstPropagated.addAndReturnNew(kMatrix.multiplyAndReturnNew(deltaZ));
492 
493         // 10. Update state estimation error covariance matrix using (3.25)
494         final var pMatrixNew = Matrix.identity(8, 8).subtractAndReturnNew(
495                 kMatrix.multiplyAndReturnNew(hMatrix)).multiplyAndReturnNew(pMatrixPropagated);
496 
497         updatedEstimation.fromMatrix(xEstNew);
498         updatedCovariance.copyFrom(pMatrixNew);
499     }
