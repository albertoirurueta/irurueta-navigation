35      @Test
36      void testConstructor() throws WrongSizeException {
37          // test empty constructor
38          var state = new GNSSKalmanState();
39  
40          // check default values
41          assertNull(state.getEstimation());
42          assertNull(state.getCovariance());
43  
44          // test constructor with estimation and covariance
45          final var randomizer = new UniformRandomizer();
46          final var x = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
47          final var y = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
48          final var z = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
49          final var vx = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
50          final var vy = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
51          final var vz = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
52          final var clockOffset = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
53          final var clockDrift = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
54          final var estimation = new GNSSEstimation(x, y, z, vx, vy, vz, clockOffset, clockDrift);
55          final var covariance = Matrix.identity(GNSSEstimation.NUM_PARAMETERS, GNSSEstimation.NUM_PARAMETERS);
56          state = new GNSSKalmanState(estimation, covariance);
57  
58          // check default values
59          assertSame(state.getEstimation(), estimation);
60          assertSame(state.getCovariance(), covariance);
61  
62          // Force IllegalArgumentException
63          final var m = new Matrix(1, 1);
64          assertThrows(IllegalArgumentException.class, () -> new GNSSKalmanState(estimation, m));
65  
66          // test copy constructor
67          state = new GNSSKalmanState(estimation, covariance);
68          final var state2 = new GNSSKalmanState(state);
69  
70          // check default values
71          assertEquals(state2.getEstimation(), estimation);
72          assertEquals(state2.getCovariance(), covariance);
73      }
