45      @Test
46      void testConstructor() {
47          // test empty constructor
48          var measurement = new GNSSMeasurement();
49  
50          // check default values
51          assertEquals(0.0, measurement.getPseudoRange(), 0.0);
52          assertEquals(0.0, measurement.getPseudoRate(), 0.0);
53          assertEquals(0.0, measurement.getX(), 0.0);
54          assertEquals(0.0, measurement.getY(), 0.0);
55          assertEquals(0.0, measurement.getZ(), 0.0);
56          assertEquals(0.0, measurement.getVx(), 0.0);
57          assertEquals(0.0, measurement.getVy(), 0.0);
58          assertEquals(0.0, measurement.getVz(), 0.0);
59  
60          // test constructor with values
61          final var randomizer = new UniformRandomizer();
62          final var pseudoRange = randomizer.nextDouble(MIN_POS_VALUE, MAX_POS_VALUE);
63          final var pseudoRate = randomizer.nextDouble(MIN_SPEED_VALUE, MAX_SPEED_VALUE);
64  
65          final var x = randomizer.nextDouble(MIN_POS_VALUE, MAX_POS_VALUE);
66          final var y = randomizer.nextDouble(MIN_POS_VALUE, MAX_POS_VALUE);
67          final var z = randomizer.nextDouble(MIN_POS_VALUE, MAX_POS_VALUE);
68  
69          final var vx = randomizer.nextDouble(MIN_SPEED_VALUE, MAX_SPEED_VALUE);
70          final var vy = randomizer.nextDouble(MIN_SPEED_VALUE, MAX_SPEED_VALUE);
71          final var vz = randomizer.nextDouble(MIN_SPEED_VALUE, MAX_SPEED_VALUE);
72  
73          measurement = new GNSSMeasurement(pseudoRange, pseudoRate, x, y, z, vx, vy, vz);
74  
75          // check default values
76          assertEquals(pseudoRange, measurement.getPseudoRange(), 0.0);
77          assertEquals(pseudoRate, measurement.getPseudoRate(), 0.0);
78          assertEquals(x, measurement.getX(), 0.0);
79          assertEquals(y, measurement.getY(), 0.0);
80          assertEquals(z, measurement.getZ(), 0.0);
81          assertEquals(vx, measurement.getVx(), 0.0);
82          assertEquals(vy, measurement.getVy(), 0.0);
83          assertEquals(vz, measurement.getVz(), 0.0);
84  
85  
86          // test constructor with measurement values
87          final var pseudoRangeDistance = new Distance(pseudoRange, DistanceUnit.METER);
88          final var pseudoRateSpeed = new Speed(pseudoRate, SpeedUnit.METERS_PER_SECOND);
89          final var distanceX = new Distance(x, DistanceUnit.METER);
90          final var distanceY = new Distance(y, DistanceUnit.METER);
91          final var distanceZ = new Distance(z, DistanceUnit.METER);
92          final var speedX = new Speed(vx, SpeedUnit.METERS_PER_SECOND);
93          final var speedY = new Speed(vy, SpeedUnit.METERS_PER_SECOND);
94          final var speedZ = new Speed(vz, SpeedUnit.METERS_PER_SECOND);
95  
96          measurement = new GNSSMeasurement(pseudoRangeDistance, pseudoRateSpeed, distanceX, distanceY, distanceZ,
97                  speedX, speedY, speedZ);
98  
99          // check default values
100         assertEquals(pseudoRange, measurement.getPseudoRange(), 0.0);
101         assertEquals(pseudoRate, measurement.getPseudoRate(), 0.0);
102         assertEquals(x, measurement.getX(), 0.0);
103         assertEquals(y, measurement.getY(), 0.0);
104         assertEquals(z, measurement.getZ(), 0.0);
105         assertEquals(vx, measurement.getVx(), 0.0);
106         assertEquals(vy, measurement.getVy(), 0.0);
107         assertEquals(vz, measurement.getVz(), 0.0);
108 
109         // test constructor with measurement values and position
110         final var position = new InhomogeneousPoint3D(x, y, z);
111 
112         measurement = new GNSSMeasurement(pseudoRangeDistance, pseudoRateSpeed, position, speedX, speedY, speedZ);
113 
114         // check default values
115         assertEquals(pseudoRange, measurement.getPseudoRange(), 0.0);
116         assertEquals(pseudoRate, measurement.getPseudoRate(), 0.0);
117         assertEquals(x, measurement.getX(), 0.0);
118         assertEquals(y, measurement.getY(), 0.0);
119         assertEquals(z, measurement.getZ(), 0.0);
120         assertEquals(vx, measurement.getVx(), 0.0);
121         assertEquals(vy, measurement.getVy(), 0.0);
122         assertEquals(vz, measurement.getVz(), 0.0);
123 
124         // test constructor with ECEF position and velocity
125         final var ecefPosition = new ECEFPosition(x, y, z);
126         final var ecefVelocity = new ECEFVelocity(vx, vy, vz);
127         measurement = new GNSSMeasurement(pseudoRange, pseudoRate, ecefPosition, ecefVelocity);
128 
129         // check default values
130         assertEquals(pseudoRange, measurement.getPseudoRange(), 0.0);
131         assertEquals(pseudoRate, measurement.getPseudoRate(), 0.0);
132         assertEquals(x, measurement.getX(), 0.0);
133         assertEquals(y, measurement.getY(), 0.0);
134         assertEquals(z, measurement.getZ(), 0.0);
135         assertEquals(vx, measurement.getVx(), 0.0);
136         assertEquals(vy, measurement.getVy(), 0.0);
137         assertEquals(vz, measurement.getVz(), 0.0);
138 
139 
140         final var positionAndVelocity = new ECEFPositionAndVelocity(ecefPosition, ecefVelocity);
141         measurement = new GNSSMeasurement(pseudoRange, pseudoRate, positionAndVelocity);
142 
143         // check default values
144         assertEquals(pseudoRange, measurement.getPseudoRange(), 0.0);
145         assertEquals(pseudoRate, measurement.getPseudoRate(), 0.0);
146         assertEquals(x, measurement.getX(), 0.0);
147         assertEquals(y, measurement.getY(), 0.0);
148         assertEquals(z, measurement.getZ(), 0.0);
149         assertEquals(vx, measurement.getVx(), 0.0);
150         assertEquals(vy, measurement.getVy(), 0.0);
151         assertEquals(vz, measurement.getVz(), 0.0);
152 
153         measurement = new GNSSMeasurement(pseudoRangeDistance, pseudoRateSpeed, ecefPosition, ecefVelocity);
154 
155         // check default values
156         assertEquals(pseudoRange, measurement.getPseudoRange(), 0.0);
157         assertEquals(pseudoRate, measurement.getPseudoRate(), 0.0);

164

166 
167         // check default values
168         assertEquals(pseudoRange, measurement.getPseudoRange(), 0.0);
169         assertEquals(pseudoRate, measurement.getPseudoRate(), 0.0);
170         assertEquals(x, measurement.getX(), 0.0);
171         assertEquals(y, measurement.getY(), 0.0);
172         assertEquals(z, measurement.getZ(), 0.0);
173         assertEquals(vx, measurement.getVx(), 0.0);
174         assertEquals(vy, measurement.getVy(), 0.0);
175         assertEquals(vz, measurement.getVz(), 0.0);
176 
177         // test copy constructor
178         final var measurement2 = new GNSSMeasurement(measurement);

182         assertEquals(measurement.getPseudoRate(), measurement2.getPseudoRate(), 0.0);
183         assertEquals(measurement.getX(), measurement2.getX(), 0.0);
