16  package com.irurueta.navigation.inertial;
17  
18  import com.irurueta.algebra.Matrix;
19  import com.irurueta.algebra.WrongSizeException;
20  import com.irurueta.statistics.UniformRandomizer;
21  import org.junit.Test;
22  
23  import java.util.Random;
24  
25  import static org.junit.Assert.assertEquals;
26  
27  public class INSLooselyCoupledKalmanInitializerTest {
28  
29      private static final double MIN_VALUE = 1e-4;
30      private static final double MAX_VALUE = 1e-3;
31  
32      private static final int NUM_PARAMS = INSLooselyCoupledKalmanInitializer.NUM_PARAMS;
33  
34      @Test
35      public void testInitialize() throws WrongSizeException {
36          final INSLooselyCoupledKalmanInitializerConfig config = generateConfig();
37  
38          final Matrix expected = Matrix.diagonal(new double[]{
39                  config.getInitialAttitudeUncertainty() * config.getInitialAttitudeUncertainty(),
40                  config.getInitialAttitudeUncertainty() * config.getInitialAttitudeUncertainty(),
41                  config.getInitialAttitudeUncertainty() * config.getInitialAttitudeUncertainty(),
42                  config.getInitialVelocityUncertainty() * config.getInitialVelocityUncertainty(),
43                  config.getInitialVelocityUncertainty() * config.getInitialVelocityUncertainty(),
44                  config.getInitialVelocityUncertainty() * config.getInitialVelocityUncertainty(),
45                  config.getInitialPositionUncertainty() * config.getInitialPositionUncertainty(),
46                  config.getInitialPositionUncertainty() * config.getInitialPositionUncertainty(),
47                  config.getInitialPositionUncertainty() * config.getInitialPositionUncertainty(),
48                  config.getInitialAccelerationBiasUncertainty() * config.getInitialAccelerationBiasUncertainty(),
49                  config.getInitialAccelerationBiasUncertainty() * config.getInitialAccelerationBiasUncertainty(),
50                  config.getInitialAccelerationBiasUncertainty() * config.getInitialAccelerationBiasUncertainty(),
51                  config.getInitialGyroscopeBiasUncertainty() * config.getInitialGyroscopeBiasUncertainty(),
52                  config.getInitialGyroscopeBiasUncertainty() * config.getInitialGyroscopeBiasUncertainty(),
53                  config.getInitialGyroscopeBiasUncertainty() * config.getInitialGyroscopeBiasUncertainty()
54          });
55  
56          final Matrix result1 = new Matrix(NUM_PARAMS, NUM_PARAMS);
57          INSLooselyCoupledKalmanInitializer.initialize(config, result1);
58          final Matrix result2 = new Matrix(1, 1);
59          INSLooselyCoupledKalmanInitializer.initialize(config, result2);
60          final Matrix result3 = INSLooselyCoupledKalmanInitializer.initialize(config);
61  
62          assertEquals(expected, result1);
63          assertEquals(expected, result2);
64          assertEquals(expected, result3);
65      }
66  
67      private static INSLooselyCoupledKalmanInitializerConfig generateConfig() {
68          final UniformRandomizer randomizer = new UniformRandomizer(new Random());
69          final double initialAttitudeUncertainty = randomizer.nextDouble(
70                  MIN_VALUE, MAX_VALUE);
71          final double initialVelocityUncertainty = randomizer.nextDouble(
72                  MIN_VALUE, MAX_VALUE);
73          final double initialPositionUncertainty = randomizer.nextDouble(
74                  MIN_VALUE, MAX_VALUE);
75          final double initialAccelerationBiasUncertainty = randomizer.nextDouble(
76                  MIN_VALUE, MAX_VALUE);
77          final double initialGyroscopeBiasUncertainty = randomizer.nextDouble(
78                  MIN_VALUE, MAX_VALUE);
79          return new INSLooselyCoupledKalmanInitializerConfig(
80                  initialAttitudeUncertainty, initialVelocityUncertainty,
81                  initialPositionUncertainty, initialAccelerationBiasUncertainty,
82                  initialGyroscopeBiasUncertainty);
83      }
84  }
