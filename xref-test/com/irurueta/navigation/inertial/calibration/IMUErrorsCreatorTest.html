251     @Test
252     public void testCreate() throws WrongSizeException,
253             InvalidSourceAndDestinationFrameTypeException, LockedException,
254             InvalidRotationMatrixException, NotReadyException {
255         final Matrix ba = generateBa();
256         final Matrix bg = generateBg();
257         final Matrix ma = generateMaCommonAxis();
258         final Matrix mg = generateMg();
259         final Matrix gg = new Matrix(3, 3);
260 
261         final double gyroNoiseRootPSD = 0.0;
262         final double accelQuantLevel = 0.0;
263         final double gyroQuantLevel = 0.0;
264 
265         final IMUErrors errors = new IMUErrors(ba, bg, ma, mg, gg, SMALL_ROOT_PSD,
266                 gyroNoiseRootPSD, accelQuantLevel, gyroQuantLevel);
267 
268         int numValid = 0;
269         for (int t = 0; t < TIMES; t++) {
270             final Random random = new Random();
271             final UniformRandomizer randomizer = new UniformRandomizer(random);
272             final double latitude = Math.toRadians(
273                     randomizer.nextDouble(MIN_LATITUDE_DEGREES, MAX_LATITUDE_DEGREES));
274             final double longitude = Math.toRadians(
275                     randomizer.nextDouble(MIN_LONGITUDE_DEGREES, MAX_LONGITUDE_DEGREES));
276             final double height = randomizer.nextDouble(MIN_HEIGHT, MAX_HEIGHT);
277             final NEDPosition nedPosition = new NEDPosition(latitude, longitude, height);
278 
279             final double roll = Math.toRadians(
280                     randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
281             final double pitch = Math.toRadians(
282                     randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
283             final double yaw = Math.toRadians(
284                     randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
285             final CoordinateTransformation nedC = new CoordinateTransformation(
286                     roll, pitch, yaw, FrameType.BODY_FRAME,
287                     FrameType.LOCAL_NAVIGATION_FRAME);
288 
289             final NEDFrame nedFrame = new NEDFrame(nedPosition, nedC);
290             final ECEFFrame ecefFrame = NEDtoECEFFrameConverter
291                     .convertNEDtoECEFAndReturnNew(nedFrame);
292 
293             // compute ground-truth kinematics that should be generated at provided
294             // position, velocity and orientation
295             final BodyKinematics trueKinematics = ECEFKinematicsEstimator
296                     .estimateKinematicsAndReturnNew(TIME_INTERVAL_SECONDS,
297                             ecefFrame, ecefFrame);
298 
299             final List<BodyKinematicsSequence<StandardDeviationTimedBodyKinematics>> gyroscopeMeasurements =
300                     new ArrayList<>();
301             final List<StandardDeviationBodyKinematics> accelerometerMeasurements =
302                     new ArrayList<>();
303 
304             final AccelerometerAndGyroscopeMeasurementsGenerator generator =
305                     new AccelerometerAndGyroscopeMeasurementsGenerator(
306                             new AccelerometerAndGyroscopeMeasurementsGeneratorListener() {
307                                 @Override
308                                 public void onInitializationStarted(
309                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
310 
311                                 }
312 
313                                 @Override
314                                 public void onInitializationCompleted(
315                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator,
316                                         final double accelerometerBaseNoiseLevel) {
317 
318                                 }
319 
320                                 @Override
321                                 public void onError(
322                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator,
323                                         final TriadStaticIntervalDetector.ErrorReason reason) {
324 
325                                 }
326 
327                                 @Override
328                                 public void onStaticIntervalDetected(
329                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
330 
331                                 }
332 
333                                 @Override
334                                 public void onDynamicIntervalDetected(
335                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
336 
337                                 }
338 
339                                 @Override
340                                 public void onStaticIntervalSkipped(
341                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
342 
343                                 }
344 
345                                 @Override
346                                 public void onDynamicIntervalSkipped(
347                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
348 
349                                 }
350

352                                 public void onGeneratedAccelerometerMeasurement(
353                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator,
354                                         final StandardDeviationBodyKinematics measurement) {
355                                     accelerometerMeasurements.add(measurement);
356                                 }
357 
358                                 @Override
359                                 public void onGeneratedGyroscopeMeasurement(
360                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator,
361                                         final BodyKinematicsSequence<StandardDeviationTimedBodyKinematics> measurement) {
362                                     gyroscopeMeasurements.add(measurement);
363                                 }
364 
365                                 @Override
366                                 public void onReset(
367                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
368 
369                                 }
370                             });
371 
372             // generate initial static samples
373             final int initialStaticSamples = TriadStaticIntervalDetector
374                     .DEFAULT_INITIAL_STATIC_SAMPLES;
375             generateStaticSamples(generator, initialStaticSamples, trueKinematics,
376                     errors, random, 0);
377 
378             final int numSequences = EasyGyroscopeCalibrator.MINIMUM_SEQUENCES_COMMON_Z_AXIS;
379             final int numMeasurements = KnownGravityNormAccelerometerCalibrator.MINIMUM_MEASUREMENTS_GENERAL;
380             final int n = Math.max(numSequences + 1, numMeasurements);
381 
382             final int staticPeriodLength = 3 * TriadStaticIntervalDetector.DEFAULT_WINDOW_SIZE;
383             final int dynamicPeriodLength = TriadStaticIntervalDetector.DEFAULT_WINDOW_SIZE;
384 
385             int start = initialStaticSamples;
386             for (int i = 0; i < n; i++) {
387                 // generate static samples
388                 generateStaticSamples(generator, staticPeriodLength, trueKinematics,
389                         errors, random, start);
390                 start += staticPeriodLength;
391 
392                 // generate dynamic samples
393                 generateDynamicSamples(generator, dynamicPeriodLength,
394                         trueKinematics, randomizer, ecefFrame, nedFrame,
395                         errors, random, start, false);
396                 start += dynamicPeriodLength;
397             }
398 
399             // as an initial value for gyroscope bias we can use the average
400             // gyroscope values during initialization. A more accurate initial
401             // guess for bias could be obtained by using leveling with magnetometer
402             // and accelerometer readings (once both magnetometer and accelerometer
403             // are calibrated).
404             final AngularSpeedTriad initialAvgAngularSpeed = generator.getInitialAvgAngularSpeedTriad();
405             final Matrix initialBg = initialAvgAngularSpeed.getValuesAsMatrix();
406 
407             if (generator.getStatus() == TriadStaticIntervalDetector.Status.FAILED) {
408                 continue;
409             }
410 
411             final Matrix initialMg = new Matrix(3, 3);
412             final Matrix initialGg = new Matrix(3, 3);
413             final EasyGyroscopeCalibrator gyroCalibrator =
414                     new EasyGyroscopeCalibrator(gyroscopeMeasurements,
415                             true, false,
416                             initialBg, initialMg, initialGg, ba, ma);
417

426             final Matrix estimatedGg = gyroCalibrator.getEstimatedGg();
427

435

437                 accelerometerCalibrator.calibrate();
438             } catch (final CalibrationException e) {
439                 continue;
440             }

442             final double[] estimatedBa = accelerometerCalibrator.getEstimatedBiases();
443             final Matrix estimatedMa = accelerometerCalibrator.getEstimatedMa();
444 
445             final double estimatedAccelerometerNoiseRootPsd =
446                     generator.getAccelerometerBaseNoiseLevelRootPsd();
447             final double estimatedGyroNoiseRootPsd =
448                     generator.getGyroscopeBaseNoiseLevelRootPsd();
449 
450             assertNotNull(estimatedBa);
451             assertNotNull(estimatedMa);
452             assertNotNull(estimatedBg);
453             assertNotNull(estimatedMg);
454             assertNotNull(estimatedGg);
455 
456             final double accelerometerQuantizationLevel = random.nextDouble();
457             final double gyroscopeQuantizationLevel = random.nextDouble();
458 
459             final IMUErrorsCreator creator = new IMUErrorsCreator(
460                     accelerometerCalibrator, gyroCalibrator,
461                     generator, generator, accelerometerQuantizationLevel,
462                     gyroscopeQuantizationLevel);
463 
464             final IMUErrors estimatedErrors = creator.create();
465 
466             assertNotNull(estimatedErrors);
467             assertArrayEquals(estimatedBa,
468                     estimatedErrors.getAccelerometerBiases(), 0.0);
469             assertArrayEquals(estimatedBg,
470                     estimatedErrors.getGyroBiases(), 0.0);
471             assertEquals(estimatedMa,
472                     estimatedErrors.getAccelerometerScaleFactorAndCrossCouplingErrors());
473             assertEquals(estimatedMg,
474                     estimatedErrors.getGyroScaleFactorAndCrossCouplingErrors());
475             assertEquals(estimatedGg,
476                     estimatedErrors.getGyroGDependentBiases());
477             assertEquals(estimatedAccelerometerNoiseRootPsd,
478                     estimatedErrors.getAccelerometerNoiseRootPSD(), 0.0);
479             assertEquals(estimatedGyroNoiseRootPsd,
480                     estimatedErrors.getGyroNoiseRootPSD(), 0.0);
481             assertEquals(accelerometerQuantizationLevel,
482                     estimatedErrors.getAccelerometerQuantizationLevel(), 0.0);
483             assertEquals(gyroscopeQuantizationLevel,
484                     estimatedErrors.getGyroQuantizationLevel(), 0.0);
485 
486             if (!bg.equals(gyroCalibrator.getEstimatedBiasesAsMatrix(), ABSOLUTE_ERROR)) {
487                 continue;
488             }
489             if (!mg.equals(estimatedMg, ABSOLUTE_ERROR)) {
490                 continue;
491             }
492             if (!gg.equals(estimatedGg, 0.0)) {
493                 continue;
494             }
495             assertTrue(bg.equals(gyroCalibrator.getEstimatedBiasesAsMatrix(), ABSOLUTE_ERROR));
496             assertTrue(mg.equals(estimatedMg, ABSOLUTE_ERROR));
497             assertTrue(gg.equals(estimatedGg, 0.0));
498 
499             if (!ba.equals(accelerometerCalibrator.getEstimatedBiasesAsMatrix(), SMALL_ABSOLUTE_ERROR)) {
500                 continue;
501             }
502             if (!ma.equals(estimatedMa, SMALL_ABSOLUTE_ERROR)) {
503                 continue;
504             }
505             assertTrue(ba.equals(accelerometerCalibrator.getEstimatedBiasesAsMatrix(), SMALL_ABSOLUTE_ERROR));
506             assertTrue(ma.equals(estimatedMa, SMALL_ABSOLUTE_ERROR));
507 
508             assertTrue(estimatedAccelerometerNoiseRootPsd > 0.0);
509             assertTrue(estimatedGyroNoiseRootPsd > 0.0);
510 
511             final double accelerometerNoiseLevel = generator.getAccelerometerBaseNoiseLevel();
512             final double gyroNoiseLevel = generator.getGyroscopeBaseNoiseLevel();
513             final double sqrtTimeInterval = Math.sqrt(generator.getTimeInterval());
514             assertEquals(accelerometerNoiseLevel * sqrtTimeInterval,
515                     estimatedAccelerometerNoiseRootPsd,
516                     SMALL_ABSOLUTE_ERROR);
517             assertEquals(gyroNoiseLevel * sqrtTimeInterval,
518                     estimatedGyroNoiseRootPsd,
519                     SMALL_ABSOLUTE_ERROR);
520 
521             assertEquals(SMALL_ROOT_PSD, estimatedAccelerometerNoiseRootPsd,
522                     SMALL_ABSOLUTE_ERROR);
523             assertEquals(0.0, estimatedGyroNoiseRootPsd,
524                     SMALL_ABSOLUTE_ERROR);
525 
526             numValid++;
527             break;
528         }
529 
530         assertTrue(numValid > 0);
531 
532         // Force NotReadyException
533         final IMUErrorsCreator creator = new IMUErrorsCreator();
534         try {
535             creator.create();
536             fail("NotReadyException expected but not thrown");
537         } catch (final NotReadyException ignore) {
538         }
539     }

585     private void generateStaticSamples(
586             final AccelerometerAndGyroscopeMeasurementsGenerator generator,
587             final int numSamples,
588             final BodyKinematics trueKinematics,
589             final IMUErrors errors,
590             final Random random,
591             final int startSample)
592             throws LockedException {
593 
594         final TimedBodyKinematics timedMeasuredKinematics = new TimedBodyKinematics();
595         final BodyKinematics measuredKinematics = new BodyKinematics();
596         for (int i = 0, j = startSample; i < numSamples; i++, j++) {
597 
598             BodyKinematicsGenerator.generate(TIME_INTERVAL_SECONDS,
599                     trueKinematics, errors, random, measuredKinematics);
600 
601             timedMeasuredKinematics.setKinematics(measuredKinematics);
602             timedMeasuredKinematics.setTimestampSeconds(
603                     j * TIME_INTERVAL_SECONDS);
604 
605             assertTrue(generator.process(timedMeasuredKinematics));
606         }
607     }

616             final NEDFrame nedFrame,
617             final IMUErrors errors,
618             final Random random,
619             final int startSample,
620             final boolean changePosition)
621             throws InvalidSourceAndDestinationFrameTypeException, LockedException, InvalidRotationMatrixException {
622 
623         final double sqrtTimeInterval = Math.sqrt(TIME_INTERVAL_SECONDS);
624         final double specificForceStandardDeviation = getAccelNoiseRootPSD() / sqrtTimeInterval;
625         final double angularRateStandardDeviation = getGyroNoiseRootPSD() / sqrtTimeInterval;
626 
627         final double deltaX = changePosition ?
628                 randomizer.nextDouble(MIN_DELTA_POS_METERS, MAX_DELTA_POS_METERS) : 0.0;
629         final double deltaY = changePosition ?
630                 randomizer.nextDouble(MIN_DELTA_POS_METERS, MAX_DELTA_POS_METERS) : 0.0;
631         final double deltaZ = changePosition ?
632                 randomizer.nextDouble(MIN_DELTA_POS_METERS, MAX_DELTA_POS_METERS) : 0.0;
633 
634         final double deltaRoll = Math.toRadians(randomizer.nextDouble(
635                 MIN_DELTA_ANGLE_DEGREES, MAX_DELTA_ANGLE_DEGREES));
636         final double deltaPitch = Math.toRadians(randomizer.nextDouble(
637                 MIN_DELTA_ANGLE_DEGREES, MAX_DELTA_ANGLE_DEGREES));
638         final double deltaYaw = Math.toRadians(randomizer.nextDouble(
639                 MIN_DELTA_ANGLE_DEGREES, MAX_DELTA_ANGLE_DEGREES));
640 
641         final double ecefX = ecefFrame.getX();
642         final double ecefY = ecefFrame.getY();
643         final double ecefZ = ecefFrame.getZ();
644 
645         final CoordinateTransformation nedC = nedFrame.getCoordinateTransformation();
646 
647         final double roll = nedC.getRollEulerAngle();
648         final double pitch = nedC.getPitchEulerAngle();
649         final double yaw = nedC.getYawEulerAngle();
650 
651         final Quaternion beforeQ = new Quaternion();
652         nedC.asRotation(beforeQ);
653 
654         NEDFrame oldNedFrame = new NEDFrame(nedFrame);
655         NEDFrame newNedFrame = new NEDFrame();
656         ECEFFrame oldEcefFrame = new ECEFFrame(ecefFrame);
657         ECEFFrame newEcefFrame = new ECEFFrame();
658 
659         double oldEcefX = ecefX - deltaX;
660         double oldEcefY = ecefY - deltaY;
661         double oldEcefZ = ecefZ - deltaZ;
662         double oldRoll = roll - deltaRoll;
663         double oldPitch = pitch - deltaPitch;
664         double oldYaw = yaw - deltaYaw;
665 
666         final BodyKinematicsSequence<StandardDeviationTimedBodyKinematics> trueSequence =
667                 new BodyKinematicsSequence<>();
668         final List<StandardDeviationTimedBodyKinematics> trueTimedKinematicsList =
669                 new ArrayList<>();
670 
671         final TimedBodyKinematics timedMeasuredKinematics = new TimedBodyKinematics();
672         final BodyKinematics measuredKinematics = new BodyKinematics();
673 
674         for (int i = 0, j = startSample; i < numSamples; i++, j++) {
675             final double progress = (double) i / (double) numSamples;
676 
677             final double newRoll = oldRoll + interpolate(deltaRoll, progress);
678             final double newPitch = oldPitch + interpolate(deltaPitch, progress);
679             final double newYaw = oldYaw + interpolate(deltaYaw, progress);
680             final CoordinateTransformation newNedC =
681                     new CoordinateTransformation(
682                             newRoll, newPitch, newYaw,
683                             FrameType.BODY_FRAME,
684                             FrameType.LOCAL_NAVIGATION_FRAME);
685             final NEDPosition newNedPosition = oldNedFrame.getPosition();
686 
687             newNedFrame.setPosition(newNedPosition);
688             newNedFrame.setCoordinateTransformation(newNedC);
689 
690             NEDtoECEFFrameConverter.convertNEDtoECEF(newNedFrame, newEcefFrame);
691 
692             final double newEcefX = oldEcefX + interpolate(deltaX, progress);
693             final double newEcefY = oldEcefY + interpolate(deltaY, progress);
694             final double newEcefZ = oldEcefZ + interpolate(deltaZ, progress);
695 
696             newEcefFrame.setCoordinates(newEcefX, newEcefY, newEcefZ);
697 
698             ECEFtoNEDFrameConverter.convertECEFtoNED(newEcefFrame, newNedFrame);
699 
700             final double timestampSeconds = j * TIME_INTERVAL_SECONDS;
701 
702             // update true kinematics using new position and rotation
703             ECEFKinematicsEstimator.estimateKinematics(TIME_INTERVAL_SECONDS,
704                     newEcefFrame, oldEcefFrame, trueKinematics);
705 
706             // add error to true kinematics
707             BodyKinematicsGenerator.generate(TIME_INTERVAL_SECONDS,
708                     trueKinematics, errors, random, measuredKinematics);
709 
710             timedMeasuredKinematics.setKinematics(measuredKinematics);
711             timedMeasuredKinematics.setTimestampSeconds(timestampSeconds);
712 
713             assertTrue(generator.process(timedMeasuredKinematics));
714 
715             final StandardDeviationTimedBodyKinematics trueTimedKinematics =
716                     new StandardDeviationTimedBodyKinematics(
717                             new BodyKinematics(trueKinematics), timestampSeconds,
718                             specificForceStandardDeviation,
719                             angularRateStandardDeviation);
720             trueTimedKinematicsList.add(trueTimedKinematics);
721 
722             oldNedFrame.copyFrom(newNedFrame);
723             oldEcefFrame.copyFrom(newEcefFrame);
724             oldRoll = newRoll;
725             oldPitch = newPitch;
726             oldYaw = newYaw;
727             oldEcefX = newEcefX;
728             oldEcefY = newEcefY;
729             oldEcefZ = newEcefZ;
730         }
731 
732         trueSequence.setItems(trueTimedKinematicsList);
733 
734         final Quaternion afterQ = new Quaternion();
735         QuaternionIntegrator.integrateGyroSequence(
736                 trueSequence, beforeQ, afterQ);
737 
738         final CoordinateTransformation newNedC =
739                 new CoordinateTransformation(
740                         afterQ.asInhomogeneousMatrix(),
741                         FrameType.BODY_FRAME,
742                         FrameType.LOCAL_NAVIGATION_FRAME);
743         newNedFrame.setCoordinateTransformation(newNedC);
744 
745         NEDtoECEFFrameConverter.convertNEDtoECEF(newNedFrame, newEcefFrame);
746 
747 
748         // update current ECEF and NED frames
749         ecefFrame.copyFrom(newEcefFrame);
750         nedFrame.copyFrom(newNedFrame);
751 
752         // after dynamic sequence finishes, update true kinematics for a
753         // static sequence at current frame
754         ECEFKinematicsEstimator.estimateKinematics(TIME_INTERVAL_SECONDS,
755                 newEcefFrame, newEcefFrame, trueKinematics);
756     }
757 
758     // This is required to simulate a smooth transition of values during
759     // dynamic period, to avoid a sudden rotation or translation and simulate
760     // a more natural behaviour.
761     private double interpolate(final double value, final double progress) {
762         return -2.0 * (Math.abs(progress - 0.5) - 0.5) * value;
763     }
