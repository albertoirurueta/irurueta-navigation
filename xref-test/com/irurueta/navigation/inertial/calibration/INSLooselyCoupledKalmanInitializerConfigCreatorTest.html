228     @Test
229     public void testCreate() throws AlgebraException,
230             InvalidSourceAndDestinationFrameTypeException, LockedException,
231             InvalidRotationMatrixException, NotReadyException,
232             RandomWalkEstimationException {
233         final Matrix ba = generateBa();
234         final Matrix bg = generateBg();
235         final Matrix ma = generateMaCommonAxis();
236         final Matrix mg = generateMg();
237         final Matrix gg = new Matrix(3, 3);
238 
239         final double accelNoiseRootPSD = getAccelNoiseRootPSD();
240         final double gyroNoiseRootPSD = getGyroNoiseRootPSD();
241         final double accelQuantLevel = 0.0;
242         final double gyroQuantLevel = 0.0;
243 
244         final IMUErrors errors = new IMUErrors(ba, bg, ma, mg, gg, accelNoiseRootPSD,
245                 gyroNoiseRootPSD, accelQuantLevel, gyroQuantLevel);
246 
247         int numValid = 0;
248         for (int t = 0; t < TIMES; t++) {
249             final Random random = new Random();
250             final UniformRandomizer randomizer = new UniformRandomizer(random);
251             final double latitude = Math.toRadians(
252                     randomizer.nextDouble(MIN_LATITUDE_DEGREES, MAX_LATITUDE_DEGREES));
253             final double longitude = Math.toRadians(
254                     randomizer.nextDouble(MIN_LONGITUDE_DEGREES, MAX_LONGITUDE_DEGREES));
255             final double height = randomizer.nextDouble(MIN_HEIGHT, MAX_HEIGHT);
256             final NEDPosition nedPosition = new NEDPosition(latitude, longitude, height);
257 
258             final double roll = Math.toRadians(
259                     randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
260             final double pitch = Math.toRadians(
261                     randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
262             final double yaw = Math.toRadians(
263                     randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
264             final CoordinateTransformation nedC = new CoordinateTransformation(
265                     roll, pitch, yaw, FrameType.BODY_FRAME,
266                     FrameType.LOCAL_NAVIGATION_FRAME);
267 
268             final NEDFrame nedFrame = new NEDFrame(nedPosition, nedC);
269             final ECEFFrame ecefFrame = NEDtoECEFFrameConverter
270                     .convertNEDtoECEFAndReturnNew(nedFrame);
271 
272             // compute ground-truth kinematics that should be generated at provided
273             // position, velocity and orientation
274             final BodyKinematics trueKinematics = ECEFKinematicsEstimator
275                     .estimateKinematicsAndReturnNew(TIME_INTERVAL_SECONDS,
276                             ecefFrame, ecefFrame);
277 
278             final List<BodyKinematicsSequence<StandardDeviationTimedBodyKinematics>> gyroscopeMeasurements =
279                     new ArrayList<>();
280             final List<StandardDeviationBodyKinematics> accelerometerMeasurements =
281                     new ArrayList<>();
282 
283             final AccelerometerAndGyroscopeMeasurementsGenerator generator =
284                     new AccelerometerAndGyroscopeMeasurementsGenerator(
285                             new AccelerometerAndGyroscopeMeasurementsGeneratorListener() {
286                                 @Override
287                                 public void onInitializationStarted(
288                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
289 
290                                 }
291 
292                                 @Override
293                                 public void onInitializationCompleted(
294                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator,
295                                         final double accelerometerBaseNoiseLevel) {
296 
297                                 }
298 
299                                 @Override
300                                 public void onError(
301                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator,
302                                         final TriadStaticIntervalDetector.ErrorReason reason) {
303 
304                                 }
305 
306                                 @Override
307                                 public void onStaticIntervalDetected(
308                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
309 
310                                 }
311 
312                                 @Override
313                                 public void onDynamicIntervalDetected(
314                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
315 
316                                 }
317 
318                                 @Override
319                                 public void onStaticIntervalSkipped(
320                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
321 
322                                 }
323 
324                                 @Override
325                                 public void onDynamicIntervalSkipped(
326                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
327 
328                                 }
329 
330                                 @Override
331                                 public void onGeneratedAccelerometerMeasurement(
332                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator,
333                                         final StandardDeviationBodyKinematics measurement) {
334                                     accelerometerMeasurements.add(measurement);
335                                 }
336 
337                                 @Override
338                                 public void onGeneratedGyroscopeMeasurement(
339                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator,
340                                         final BodyKinematicsSequence<StandardDeviationTimedBodyKinematics> measurement) {
341                                     gyroscopeMeasurements.add(measurement);
342                                 }
343 
344                                 @Override
345                                 public void onReset(
346                                         final AccelerometerAndGyroscopeMeasurementsGenerator generator) {
347 
348                                 }
349                             });
350 
351             // generate initial static samples
352             final int initialStaticSamples = TriadStaticIntervalDetector
353                     .DEFAULT_INITIAL_STATIC_SAMPLES;
354             generateStaticSamples(generator, initialStaticSamples, trueKinematics,
355                     errors, random, 0);
356 
357             final int numSequences = EasyGyroscopeCalibrator.MINIMUM_SEQUENCES_COMMON_Z_AXIS;
358             final int numMeasurements = KnownGravityNormAccelerometerCalibrator.MINIMUM_MEASUREMENTS_GENERAL;
359             final int n = Math.max(numSequences + 1, numMeasurements);
360 
361             final int staticPeriodLength = 3 * TriadStaticIntervalDetector.DEFAULT_WINDOW_SIZE;
362             final int dynamicPeriodLength = TriadStaticIntervalDetector.DEFAULT_WINDOW_SIZE;
363 
364             int start = initialStaticSamples;
365             for (int i = 0; i < n; i++) {
366                 // generate static samples
367                 generateStaticSamples(generator, staticPeriodLength, trueKinematics,
368                         errors, random, start);
369                 start += staticPeriodLength;
370 
371                 // generate dynamic samples
372                 generateDynamicSamples(generator, dynamicPeriodLength,
373                         trueKinematics, randomizer, ecefFrame, nedFrame,
374                         errors, random, start, false);
375                 start += dynamicPeriodLength;
376             }
377 
378             // as an initial value for gyroscope bias we can use the average
379             // gyroscope values during initialization. A more accurate initial
380             // guess for bias could be obtained by using leveling with magnetometer
381             // and accelerometer readings (once both magnetometer and accelerometer
382             // are calibrated).
383             final AngularSpeedTriad initialAvgAngularSpeed = generator.getInitialAvgAngularSpeedTriad();
384             final Matrix initialBg = initialAvgAngularSpeed.getValuesAsMatrix();
385 
386             if (generator.getStatus() == TriadStaticIntervalDetector.Status.FAILED) {
387                 continue;
388             }
389 
390             final Matrix initialMg = new Matrix(3, 3);
391             final Matrix initialGg = new Matrix(3, 3);
392             final EasyGyroscopeCalibrator gyroCalibrator =
393                     new EasyGyroscopeCalibrator(gyroscopeMeasurements,
394                             true, false,
395                             initialBg, initialMg, initialGg, ba, ma);
396 
397             try {
398                 gyroCalibrator.calibrate();
399             } catch (final CalibrationException e) {
400                 continue;
401             }
402 
403             final double[] estimatedBg = gyroCalibrator.getEstimatedBiases();
404             final Matrix estimatedMg = gyroCalibrator.getEstimatedMg();
405             final Matrix estimatedGg = gyroCalibrator.getEstimatedGg();
406 
407             final ECEFGravity gravity = ECEFGravityEstimator
408                     .estimateGravityAndReturnNew(ecefFrame);
409 
410             KnownGravityNormAccelerometerCalibrator accelerometerCalibrator =
411                     new KnownGravityNormAccelerometerCalibrator(
412                             gravity.getNorm(), accelerometerMeasurements,
413                             true);
414 
415             try {
416                 accelerometerCalibrator.calibrate();
417             } catch (final CalibrationException e) {
418                 continue;
419             }
420 
421             final double[] estimatedBa = accelerometerCalibrator.getEstimatedBiases();
422             final Matrix estimatedMa = accelerometerCalibrator.getEstimatedMa();
423 
424             assertNotNull(estimatedBa);
425             assertNotNull(estimatedMa);
426             assertNotNull(estimatedBg);
427             assertNotNull(estimatedMg);
428             assertNotNull(estimatedGg);
429 
430             final RandomWalkEstimator randomWalkEstimator = new RandomWalkEstimator();
431             randomWalkEstimator.setNedPositionAndNedOrientation(nedPosition, nedC);
432             randomWalkEstimator.setAccelerationBias(estimatedBa);
433             randomWalkEstimator.setAccelerationCrossCouplingErrors(estimatedMa);
434             randomWalkEstimator.setAngularSpeedBias(estimatedBg);
435             randomWalkEstimator.setAngularSpeedCrossCouplingErrors(estimatedMg);
436             randomWalkEstimator.setAngularSpeedGDependantCrossBias(estimatedGg);
437             randomWalkEstimator.setTimeInterval(TIME_INTERVAL_SECONDS);
438 
439             generateStaticSamples(randomWalkEstimator,
440                     trueKinematics, errors, random);
441 
442             final double attitudeStd = randomWalkEstimator.getAttitudeStandardDeviation();
443             final double velocityStd = randomWalkEstimator.getVelocityStandardDeviation();
444             final double positionStd = randomWalkEstimator.getPositionStandardDeviation();
445             final double accelerometerBiasStd = accelerometerCalibrator.getEstimatedBiasStandardDeviationNorm();
446             final double gyroBiasStd = gyroCalibrator.getEstimatedBiasStandardDeviationNorm();
447 
448             assertTrue(attitudeStd > 0.0);
449             assertTrue(velocityStd > 0.0);
450             assertTrue(positionStd > 0.0);
451             assertTrue(accelerometerBiasStd > 0.0);
452             assertTrue(gyroBiasStd > 0.0);
453 
454             final INSLooselyCoupledKalmanInitializerConfigCreator creator =
455                     new INSLooselyCoupledKalmanInitializerConfigCreator(
456                             accelerometerCalibrator, gyroCalibrator, randomWalkEstimator);
457 
458             assertTrue(creator.isReady());
459 
460             final INSLooselyCoupledKalmanInitializerConfig config = creator.create();
461 
462             assertEquals(attitudeStd, config.getInitialAttitudeUncertainty(),
463                     0.0);
464             assertEquals(velocityStd, config.getInitialVelocityUncertainty(),
465                     0.0);
466             assertEquals(positionStd, config.getInitialPositionUncertainty(),
467                     0.0);
468             assertEquals(accelerometerBiasStd,
469                     config.getInitialAccelerationBiasUncertainty(), 0.0);
470             assertEquals(gyroBiasStd,
471                     config.getInitialGyroscopeBiasUncertainty(), 0.0);
472 
473             numValid++;
474             break;
475         }
476 
477         assertTrue(numValid > 0);
478 
479         // Force NotReadyException
480         final INSLooselyCoupledKalmanInitializerConfigCreator creator =
481                 new INSLooselyCoupledKalmanInitializerConfigCreator();
482         try {
483             creator.create();
484             fail("NotReadyException expected but not thrown");
485         } catch (final NotReadyException ignore) {
486         }
487     }

533     private void generateStaticSamples(
534             final RandomWalkEstimator randomWalkEstimator,
535             final BodyKinematics trueKinematics,
536             final IMUErrors errors,
537             final Random random)
538             throws LockedException, RandomWalkEstimationException {
539 
540         final BodyKinematics measuredKinematics = new BodyKinematics();
541         for (int i = 0, j = 0; i < 5000; i++, j++) {
542 
543             BodyKinematicsGenerator.generate(TIME_INTERVAL_SECONDS,
544                     trueKinematics, errors, random, measuredKinematics);
545 
546             randomWalkEstimator.addBodyKinematics(measuredKinematics);
547         }
548     }
549 
550     private void generateStaticSamples(
551             final AccelerometerAndGyroscopeMeasurementsGenerator generator,
552             final int numSamples,
553             final BodyKinematics trueKinematics,
554             final IMUErrors errors,
555             final Random random,
556             final int startSample)
557             throws LockedException {
558 
559         final TimedBodyKinematics timedMeasuredKinematics = new TimedBodyKinematics();
560         final BodyKinematics measuredKinematics = new BodyKinematics();
561         for (int i = 0, j = startSample; i < numSamples; i++, j++) {
562 
563             BodyKinematicsGenerator.generate(TIME_INTERVAL_SECONDS,
564                     trueKinematics, errors, random, measuredKinematics);
565 
566             timedMeasuredKinematics.setKinematics(measuredKinematics);
567             timedMeasuredKinematics.setTimestampSeconds(
568                     j * TIME_INTERVAL_SECONDS);
569 
570             assertTrue(generator.process(timedMeasuredKinematics));
571         }
572     }

574     @SuppressWarnings("SameParameterValue")
575     private void generateDynamicSamples(
576             final AccelerometerAndGyroscopeMeasurementsGenerator generator,
577             final int numSamples,
578             final BodyKinematics trueKinematics,
579             final UniformRandomizer randomizer,
580             final ECEFFrame ecefFrame,
581             final NEDFrame nedFrame,
582             final IMUErrors errors,
583             final Random random,
584             final int startSample,
585             final boolean changePosition)
586             throws InvalidSourceAndDestinationFrameTypeException, LockedException, InvalidRotationMatrixException {
587 
588         final double sqrtTimeInterval = Math.sqrt(TIME_INTERVAL_SECONDS);
589         final double specificForceStandardDeviation = getAccelNoiseRootPSD() / sqrtTimeInterval;
590         final double angularRateStandardDeviation = getGyroNoiseRootPSD() / sqrtTimeInterval;
591 
592         final double deltaX = changePosition ?
593                 randomizer.nextDouble(MIN_DELTA_POS_METERS, MAX_DELTA_POS_METERS) : 0.0;
594         final double deltaY = changePosition ?
595                 randomizer.nextDouble(MIN_DELTA_POS_METERS, MAX_DELTA_POS_METERS) : 0.0;
596         final double deltaZ = changePosition ?
597                 randomizer.nextDouble(MIN_DELTA_POS_METERS, MAX_DELTA_POS_METERS) : 0.0;
598 
599         final double deltaRoll = Math.toRadians(randomizer.nextDouble(
600                 MIN_DELTA_ANGLE_DEGREES, MAX_DELTA_ANGLE_DEGREES));
601         final double deltaPitch = Math.toRadians(randomizer.nextDouble(
602                 MIN_DELTA_ANGLE_DEGREES, MAX_DELTA_ANGLE_DEGREES));
603         final double deltaYaw = Math.toRadians(randomizer.nextDouble(
604                 MIN_DELTA_ANGLE_DEGREES, MAX_DELTA_ANGLE_DEGREES));
605 
606         final double ecefX = ecefFrame.getX();
607         final double ecefY = ecefFrame.getY();
608         final double ecefZ = ecefFrame.getZ();
609 
610         final CoordinateTransformation nedC = nedFrame.getCoordinateTransformation();
611 
612         final double roll = nedC.getRollEulerAngle();
613         final double pitch = nedC.getPitchEulerAngle();
614         final double yaw = nedC.getYawEulerAngle();
615 
616         final Quaternion beforeQ = new Quaternion();
617         nedC.asRotation(beforeQ);
618 
619         NEDFrame oldNedFrame = new NEDFrame(nedFrame);
620         NEDFrame newNedFrame = new NEDFrame();
621         ECEFFrame oldEcefFrame = new ECEFFrame(ecefFrame);
622         ECEFFrame newEcefFrame = new ECEFFrame();
623 
624         double oldEcefX = ecefX - deltaX;
625         double oldEcefY = ecefY - deltaY;
626         double oldEcefZ = ecefZ - deltaZ;
627         double oldRoll = roll - deltaRoll;
628         double oldPitch = pitch - deltaPitch;
629         double oldYaw = yaw - deltaYaw;
630 
631         final BodyKinematicsSequence<StandardDeviationTimedBodyKinematics> trueSequence =
632                 new BodyKinematicsSequence<>();
633         final List<StandardDeviationTimedBodyKinematics> trueTimedKinematicsList =
634                 new ArrayList<>();
635 
636         final TimedBodyKinematics timedMeasuredKinematics = new TimedBodyKinematics();
637         final BodyKinematics measuredKinematics = new BodyKinematics();
638 
639         for (int i = 0, j = startSample; i < numSamples; i++, j++) {
640             final double progress = (double) i / (double) numSamples;
641 
642             final double newRoll = oldRoll + interpolate(deltaRoll, progress);
643             final double newPitch = oldPitch + interpolate(deltaPitch, progress);
644             final double newYaw = oldYaw + interpolate(deltaYaw, progress);
645             final CoordinateTransformation newNedC =
646                     new CoordinateTransformation(
647                             newRoll, newPitch, newYaw,
648                             FrameType.BODY_FRAME,
649                             FrameType.LOCAL_NAVIGATION_FRAME);
650             final NEDPosition newNedPosition = oldNedFrame.getPosition();
651 
652             newNedFrame.setPosition(newNedPosition);
653             newNedFrame.setCoordinateTransformation(newNedC);
654 
655             NEDtoECEFFrameConverter.convertNEDtoECEF(newNedFrame, newEcefFrame);
656 
657             final double newEcefX = oldEcefX + interpolate(deltaX, progress);
658             final double newEcefY = oldEcefY + interpolate(deltaY, progress);
659             final double newEcefZ = oldEcefZ + interpolate(deltaZ, progress);
660 
661             newEcefFrame.setCoordinates(newEcefX, newEcefY, newEcefZ);
662 
663             ECEFtoNEDFrameConverter.convertECEFtoNED(newEcefFrame, newNedFrame);
664 
665             final double timestampSeconds = j * TIME_INTERVAL_SECONDS;
666 
667             // update true kinematics using new position and rotation
668             ECEFKinematicsEstimator.estimateKinematics(TIME_INTERVAL_SECONDS,
669                     newEcefFrame, oldEcefFrame, trueKinematics);
670 
671             // add error to true kinematics
672             BodyKinematicsGenerator.generate(TIME_INTERVAL_SECONDS,
673                     trueKinematics, errors, random, measuredKinematics);
674 
675             timedMeasuredKinematics.setKinematics(measuredKinematics);
676             timedMeasuredKinematics.setTimestampSeconds(timestampSeconds);
677 
678             assertTrue(generator.process(timedMeasuredKinematics));
679 
680             final StandardDeviationTimedBodyKinematics trueTimedKinematics =
681                     new StandardDeviationTimedBodyKinematics(
682                             new BodyKinematics(trueKinematics), timestampSeconds,
683                             specificForceStandardDeviation,
684                             angularRateStandardDeviation);
685             trueTimedKinematicsList.add(trueTimedKinematics);
686 
687             oldNedFrame.copyFrom(newNedFrame);
688             oldEcefFrame.copyFrom(newEcefFrame);
689             oldRoll = newRoll;
690             oldPitch = newPitch;
691             oldYaw = newYaw;
692             oldEcefX = newEcefX;
693             oldEcefY = newEcefY;
694             oldEcefZ = newEcefZ;
695         }
696 
697         trueSequence.setItems(trueTimedKinematicsList);
698 
699         final Quaternion afterQ = new Quaternion();
700         QuaternionIntegrator.integrateGyroSequence(
701                 trueSequence, beforeQ, afterQ);
702 
703         final CoordinateTransformation newNedC =
704                 new CoordinateTransformation(
705                         afterQ.asInhomogeneousMatrix(),
706                         FrameType.BODY_FRAME,
707                         FrameType.LOCAL_NAVIGATION_FRAME);
708         newNedFrame.setCoordinateTransformation(newNedC);
709 
710         NEDtoECEFFrameConverter.convertNEDtoECEF(newNedFrame, newEcefFrame);
711 
712 
713         // update current ECEF and NED frames
714         ecefFrame.copyFrom(newEcefFrame);
715         nedFrame.copyFrom(newNedFrame);
716 
717         // after dynamic sequence finishes, update true kinematics for a
718         // static sequence at current frame
719         ECEFKinematicsEstimator.estimateKinematics(TIME_INTERVAL_SECONDS,
720                 newEcefFrame, newEcefFrame, trueKinematics);
721     }
722 
723     // This is required to simulate a smooth transition of values during
724     // dynamic period, to avoid a sudden rotation or translation and simulate
725     // a more natural behaviour.
726     private double interpolate(final double value, final double progress) {
727         return -2.0 * (Math.abs(progress - 0.5) - 0.5) * value;
728     }
