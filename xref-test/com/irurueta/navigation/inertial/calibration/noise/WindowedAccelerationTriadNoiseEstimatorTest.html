399     @Test
400     public void testAddTriadAndProcessAndThenReset1() throws WrongSizeException, LockedException {
401         final Matrix ba = generateBa();
402         final Matrix bg = generateBg();
403         final Matrix ma = generateMa();
404         final Matrix mg = generateMg();
405         final Matrix gg = generateGg();
406         final double accelNoiseRootPSD = getAccelNoiseRootPsd();
407         final double gyroNoiseRootPSD = getGyroNoiseRootPsd();
408         final double accelQuantLevel = 0.0;
409         final double gyroQuantLevel = 0.0;
410 
411         final IMUErrors errors = new IMUErrors(ba, bg, ma, mg, gg, accelNoiseRootPSD,
412                 gyroNoiseRootPSD, accelQuantLevel, gyroQuantLevel);
413 
414         final Random random = new Random();
415         final UniformRandomizer randomizer = new UniformRandomizer(random);
416         final double fx = randomizer.nextDouble(MIN_ACCELEROMETER_VALUE,
417                 MAX_ACCELEROMETER_VALUE);
418         final double fy = randomizer.nextDouble(MIN_ACCELEROMETER_VALUE,
419                 MAX_ACCELEROMETER_VALUE);
420         final double fz = randomizer.nextDouble(MIN_ACCELEROMETER_VALUE,
421                 MAX_ACCELEROMETER_VALUE);
422         final double omegaX = 0.0;
423         final double omegaY = 0.0;
424         final double omegaZ = 0.0;
425 
426         final BodyKinematics trueKinematics = new BodyKinematics(fx, fy, fz,
427                 omegaX, omegaY, omegaZ);
428 
429         final WindowedAccelerationTriadNoiseEstimator estimator =
430                 new WindowedAccelerationTriadNoiseEstimator(this);
431 
432         reset();
433         assertEquals(mStart, 0);
434         assertEquals(mTriadAdded, 0);
435         assertEquals(mWindowFilled, 0);
436         assertEquals(mReset, 0);
437         assertFalse(estimator.isWindowFilled());
438         assertEquals(estimator.getNumberOfProcessedSamples(), 0);
439         assertNull(estimator.getFirstWindowedTriad());
440         assertNull(estimator.getLastWindowedTriad());
441         assertFalse(estimator.isRunning());
442 
443         final BodyKinematics kinematics = new BodyKinematics();
444         final BodyKinematics firstKinematics = new BodyKinematics();
445         final int windowSize = estimator.getWindowSize();
446         final double timeInterval = estimator.getTimeInterval();
447         final AccelerationTriad firstTriad = new AccelerationTriad();
448         final AccelerationTriad lastTriad = new AccelerationTriad();
449         final AccelerationTriad triad = new AccelerationTriad();
450         double valueX;
451         double valueY;
452         double valueZ;
453         double avgFx = 0.0;
454         double avgFy = 0.0;
455         double avgFz = 0.0;
456         final List<AccelerationTriad> triads = new ArrayList<>();
457         for (int i = 0; i < windowSize; i++) {
458             if (estimator.getFirstWindowedTriad(firstTriad)) {
459                 assertEquals(estimator.getFirstWindowedTriad(), firstTriad);
460                 assertEquals(firstTriad, firstKinematics.getSpecificForceTriad());
461             }
462             if (estimator.getLastWindowedTriad(lastTriad)) {
463                 assertEquals(estimator.getLastWindowedTriad(), lastTriad);
464                 assertEquals(lastTriad, triad);
465             }
466 
467             BodyKinematicsGenerator.generate(timeInterval, trueKinematics,
468                     errors, random, kinematics);
469 
470             if (i == 0) {
471                 firstKinematics.copyFrom(kinematics);
472             }
473 
474             kinematics.getSpecificForceTriad(triad);
475             triads.add(new AccelerationTriad(triad));
476             valueX = triad.getValueX();
477             valueY = triad.getValueY();
478             valueZ = triad.getValueZ();
479 
480             estimator.addTriadAndProcess(triad);
481 
482             assertTrue(estimator.getLastWindowedTriad(lastTriad));
483             assertEquals(lastTriad, triad);
484             assertEquals(estimator.getNumberOfProcessedSamples(), i + 1);
