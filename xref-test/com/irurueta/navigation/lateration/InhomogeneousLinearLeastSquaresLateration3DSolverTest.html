193     @Test
194     void testGetSetSpheres() throws LockedException {
195         final var randomizer = new UniformRandomizer();
196 
197         final var solver = new InhomogeneousLinearLeastSquaresLateration3DSolver();
198 
199         // initial value
200         assertNull(solver.getSpheres());
201 
202         // set new value
203         final var positions = new Point3D[4];
204         positions[0] = new InhomogeneousPoint3D(randomizer.nextDouble(), randomizer.nextDouble(),
205                 randomizer.nextDouble());
206         positions[1] = new InhomogeneousPoint3D(randomizer.nextDouble(), randomizer.nextDouble(),
207                 randomizer.nextDouble());
208         positions[2] = new InhomogeneousPoint3D(randomizer.nextDouble(), randomizer.nextDouble(),
209                 randomizer.nextDouble());
210         positions[3] = new InhomogeneousPoint3D(randomizer.nextDouble(), randomizer.nextDouble(),
211                 randomizer.nextDouble());
212         final var distances = new double[4];
213         distances[0] = randomizer.nextDouble(1.0, MAX_RANDOM_VALUE);
214         distances[1] = randomizer.nextDouble(1.0, MAX_RANDOM_VALUE);
215         distances[2] = randomizer.nextDouble(1.0, MAX_RANDOM_VALUE);
216         distances[3] = randomizer.nextDouble(1.0, MAX_RANDOM_VALUE);
217 
218         final var spheres = new Sphere[4];
219         spheres[0] = new Sphere(positions[0], distances[0]);
220         spheres[1] = new Sphere(positions[1], distances[1]);
221         spheres[2] = new Sphere(positions[2], distances[2]);
222         spheres[3] = new Sphere(positions[3], distances[3]);
223 
224         solver.setSpheres(spheres);
225 
226         // check
227         final var spheres2 = solver.getSpheres();
228         for (var i = 0; i < 4; i++) {
229             assertSame(spheres[i].getCenter(), spheres2[i].getCenter());
230             assertEquals(spheres[i].getRadius(), spheres2[i].getRadius(), 0.0);
231         }
232 
233         // force IllegalArgumentException
234         assertThrows(IllegalArgumentException.class, () -> solver.setSpheres(null));
235         assertThrows(IllegalArgumentException.class, () -> solver.setSpheres(new Sphere[1]));
236     }

286     @Test
287     void testSolveNoError() throws LaterationException, NotReadyException, LockedException {
288         final var randomizer = new UniformRandomizer();
289 
290         var numValid = 0;
291         for (var t = 0; t < TIMES; t++) {
292             final var numSpheres = randomizer.nextInt(MIN_SPHERES, MAX_SPHERES);
293 
294             final var position = new InhomogeneousPoint3D(
295                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
296                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
297                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
298             final var spheres = new Sphere[numSpheres];
299             for (var i = 0; i < numSpheres; i++) {
300                 final var center = new InhomogeneousPoint3D(
301                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
302                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
303                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
304                 final var radius = center.distanceTo(position);
305                 spheres[i] = new Sphere(center, radius);
306             }
307 
308             final var solver = new InhomogeneousLinearLeastSquaresLateration3DSolver(spheres, this);
309 
310             reset();
311             assertEquals(0, solveStart);
312             assertEquals(0, solveEnd);
313             assertTrue(solver.isReady());
314             assertNull(solver.getEstimatedPosition());
315             assertNull(solver.getEstimatedPositionCoordinates());
316 
317             solver.solve();
318 
319             final var estimatedPosition = solver.getEstimatedPosition();
320             if (estimatedPosition.distanceTo(position) > ABSOLUTE_ERROR) {
321                 continue;
322             }
323             assertTrue(position.equals(estimatedPosition, ABSOLUTE_ERROR));
324 
325             numValid++;
326             break;
327         }
328 
329         assertTrue(numValid > 0);
330 
331         // Force NotReadyException
332         final var solver = new InhomogeneousLinearLeastSquaresLateration3DSolver();
333         assertThrows(NotReadyException.class, solver::solve);
334 
335         // Force LaterationException
336         final var circles = new Sphere[4];
337         for (var i = 0; i < 4; i++) {
338             final var center = new InhomogeneousPoint3D(0.0, 0.0, 0.0);
339             final var radius = LaterationSolver.EPSILON;
340             circles[i] = new Sphere(center, radius);
341         }
342         solver.setSpheres(circles);
343         assertThrows(LaterationException.class, solver::solve);
344     }

346     @Test
347     void testSolveWithError() throws LaterationException, NotReadyException, LockedException {
348         final var randomizer = new UniformRandomizer();
349 
350         var numValid = 0;
351         var numInvalid = 0;
352         for (var t = 0; t < TIMES; t++) {
353             final var numPoints = randomizer.nextInt(MIN_SPHERES, MAX_SPHERES);
354 
355             final var position = new InhomogeneousPoint3D(
356                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
357                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
358                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
359             final var positions = new Point3D[numPoints];
360             final var distances = new double[numPoints];
361             for (var i = 0; i < numPoints; i++) {
362                 final var point = new InhomogeneousPoint3D(
363                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
364                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
365                         randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
366                 final var distance = point.distanceTo(position);
367                 final var error = randomizer.nextDouble(MIN_DISTANCE_ERROR, MAX_DISTANCE_ERROR);
368                 positions[i] = point;
369                 distances[i] = distance + error;
370             }
371 
372             final var solver = new InhomogeneousLinearLeastSquaresLateration3DSolver(positions, distances,
373                     this);
374 
375             reset();
376             assertEquals(0, solveStart);
377             assertEquals(0, solveEnd);
378             assertTrue(solver.isReady());
379             assertNull(solver.getEstimatedPosition());
380             assertNull(solver.getEstimatedPositionCoordinates());
381 
382             solver.solve();
383 
384             final var estimatedPosition = solver.getEstimatedPosition();
385             final var distance = estimatedPosition.distanceTo(position);
386             if (distance >= LARGE_ABSOLUTE_ERROR) {
387                 numInvalid++;
388             } else {
389                 numValid++;
390                 assertTrue(position.equals(estimatedPosition, LARGE_ABSOLUTE_ERROR));
391             }
392         }
393 
394         assertTrue(numValid > numInvalid);
395     }

457     @Override
458     public void onSolveStart(final LaterationSolver<Point3D> solver) {
459         solveStart++;
460         checkLocked((InhomogeneousLinearLeastSquaresLateration3DSolver) solver);
461     }
462 
463     @Override
464     public void onSolveEnd(final LaterationSolver<Point3D> solver) {
465         solveEnd++;
466         checkLocked((InhomogeneousLinearLeastSquaresLateration3DSolver) solver);
467     }
468 
469     private void reset() {
470         solveStart = solveEnd = 0;
471     }
472 
473     private static void checkLocked(final InhomogeneousLinearLeastSquaresLateration3DSolver solver) {
474         assertThrows(LockedException.class, () -> solver.setListener(null));
475         assertThrows(LockedException.class, () -> solver.setPositionsAndDistances(null, null));
476         assertThrows(LockedException.class, () -> solver.setSpheres(null));
477         assertThrows(LockedException.class, solver::solve);
