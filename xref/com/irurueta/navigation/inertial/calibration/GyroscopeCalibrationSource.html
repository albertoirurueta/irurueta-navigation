1   package com.irurueta.navigation.inertial.calibration;
2   
3   import com.irurueta.algebra.Matrix;
4   
5   /**
6    * Defines a source for estimated gyroscope calibration data.
7    */
8   public interface GyroscopeCalibrationSource {
9   
10      /**
11       * Gets array containing x,y,z components of estimated gyroscope biases
12       * expressed in radians per second (rad/s).
13       *
14       * @return array containing x,y,z components of estimated gyroscope biases.
15       */
16      double[] getEstimatedBiases();
17  
18      /**
19       * Gets estimated gyroscope scale factors and cross coupling errors.
20       * This is the product of matrix Tg containing cross coupling errors and Kg
21       * containing scaling factors.
22       * So that:
23       * <pre>
24       *     Mg = [sx    mxy  mxz] = Tg*Kg
25       *          [myx   sy   myz]
26       *          [mzx   mzy  sz ]
27       * </pre>
28       * Where:
29       * <pre>
30       *     Kg = [sx 0   0 ]
31       *          [0  sy  0 ]
32       *          [0  0   sz]
33       * </pre>
34       * and
35       * <pre>
36       *     Tg = [1          -alphaXy    alphaXz ]
37       *          [alphaYx    1           -alphaYz]
38       *          [-alphaZx   alphaZy     1       ]
39       * </pre>
40       * Hence:
41       * <pre>
42       *     Mg = [sx    mxy  mxz] = Tg*Kg =  [sx             -sy * alphaXy   sz * alphaXz ]
43       *          [myx   sy   myz]            [sx * alphaYx   sy              -sz * alphaYz]
44       *          [mzx   mzy  sz ]            [-sx * alphaZx  sy * alphaZy    sz           ]
45       * </pre>
46       * This instance allows any 3x3 matrix however, typically alphaYx, alphaZx and alphaZy
47       * are considered to be zero if the gyroscope z-axis is assumed to be the same
48       * as the body z-axis. When this is assumed, myx = mzx = mzy = 0 and the Mg matrix
49       * becomes upper diagonal:
50       * <pre>
51       *     Mg = [sx    mxy  mxz]
52       *          [0     sy   myz]
53       *          [0     0    sz ]
54       * </pre>
55       * Values of this matrix are unitless.
56       *
57       * @return estimated gyroscope scale factors and cross coupling errors.
58       */
59      Matrix getEstimatedMg();
60  
61      /**
62       * Gets estimated G-dependent cross biases introduced on the gyroscope by the
63       * specific forces sensed by the accelerometer.
64       *
65       * @return a 3x3 matrix containing g-dependent cross biases.
66       */
67      Matrix getEstimatedGg();
68  }
