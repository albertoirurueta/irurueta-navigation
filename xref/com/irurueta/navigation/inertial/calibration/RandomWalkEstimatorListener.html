5   /**
6    * Listener to handle events generated by {@link RandomWalkEstimator}
7    */
8   public interface RandomWalkEstimatorListener {
9   
10      /**
11       * Called when estimation starts.
12       *
13       * @param estimator estimator that raised the event.
14       */
15      void onStart(final RandomWalkEstimator estimator);
16  
17      /**
18       * Called when a measured body kinematics sample is added containing new
19       * measured accelerometer and gyroscope measures.
20       *
21       * @param estimator estimator that raised the event.
22       * @param measuredKinematics measured kinematics being added.
23       * @param fixedKinematics fixed kinematics taking into account calibration data.
24       */
25      void onBodyKinematicsAdded(final RandomWalkEstimator estimator,
26                                 final BodyKinematics measuredKinematics,
27                                 final BodyKinematics fixedKinematics);
28  
29      /**
30       * Called when estimation is reset.
31       *
32       * @param estimator estimator that raised the event.
33       */
34      void onReset(final RandomWalkEstimator estimator);
35  }
