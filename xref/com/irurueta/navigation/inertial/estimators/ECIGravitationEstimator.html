36       * The equatorial radius of WGS84 ellipsoid (6378137 m) defining Earth's shape.
37       */
38      public static final double EARTH_EQUATORIAL_RADIUS_WGS84 =
39              Constants.EARTH_EQUATORIAL_RADIUS_WGS84;

41      /**
42       * WGS84 Earth gravitational constant expressed in m^3 * s^-2
43       */
44      public static final double EARTH_GRAVITATIONAL_CONSTANT =
45              Constants.EARTH_GRAVITATIONAL_CONSTANT;

160     /**
161      * Estimates gravitational acceleration resolved about ECI-frame axes.
162      *
163      * @param x      cartesian x coordinate of body position expressed in meters (m) with respect ECI frame, resolved
164      *               along ECI-frame axes.
165      * @param y      cartesian y coordinate of body position expressed in meters (m) with respect ECI frame, resolved
166      *               along ECI-frame axes.
167      * @param z      cartesian z coordinate of body position expressed in meters (m) with respect ECI frame, resolved
168      *               along ECI-frame axes.
169      * @param result instance where estimated acceleration due to gravity will be stored.
170      */
171     public static void estimateGravitation(
172             final double x, final double y, final double z, final ECIGravitation result) {
173 
174         // Calculate distance from center of the Earth
175         final double magR = Math.sqrt(x * x + y * y + z * z);
176 
177         // If the input position is 0,0,0, produce a dummy output
178         if (magR == 0.0) {
179             result.setCoordinates(0.0, 0.0, 0.0);
180         } else {
181             final double zScale = 5.0 * Math.pow(z / magR, 2.0);
182             final double tmp1 = -EARTH_GRAVITATIONAL_CONSTANT / Math.pow(magR, 3.0);
183             final double tmp2 = 1.5 * EARTH_SECOND_GRAVITATIONAL_CONSTANT * Math.pow(
184                     EARTH_EQUATORIAL_RADIUS_WGS84 / magR, 2.0);
185             final double gx = tmp1 * (1.0 + tmp2 * (1.0 - zScale)) * x;
186             final double gy = tmp1 * (1.0 + tmp2 * (1.0 - zScale)) * y;
187             final double gz = tmp1 * (1.0 + tmp2 * (3.0 - zScale)) * z;
188 
189             result.setCoordinates(gx, gy, gz);
190         }
191     }
